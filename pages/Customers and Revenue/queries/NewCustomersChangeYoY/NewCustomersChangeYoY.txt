SELECT 
  x,
  (z - LAG(z) OVER (ORDER BY x)) / LAG(z) OVER (ORDER BY x) * 100 AS y,
  z
FROM (
  SELECT 
    DATE_FORMAT(date_created, '%M %Y') AS x, 
    COUNT(CASE 
            WHEN YEAR(date_created) = YEAR(CURRENT_DATE - INTERVAL 1 YEAR) AND MONTH(date_created) = MONTH(CURRENT_DATE) AND DAY(date_created) <= DAY(CURRENT_DATE) THEN 1
            WHEN YEAR(date_created) <> YEAR(CURRENT_DATE - INTERVAL 1 YEAR) OR MONTH(date_created) <> MONTH(CURRENT_DATE) THEN 1
            ELSE NULL
          END) AS z,
    MIN(date_created) AS min_date
  FROM 
    wp_edd_customers
  GROUP BY 
    MONTH(date_created), 
    YEAR(date_created)
  ORDER BY 
    date_created
) subq
WHERE z IS NOT NULL
ORDER BY min_date;
